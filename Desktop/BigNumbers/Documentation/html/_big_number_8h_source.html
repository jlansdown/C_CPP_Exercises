208  static vector<int> convert_base(const vector<int>& rhs, int prev, int newDigits) {
209  vector<long long> p(max(prev, newDigits) + 1);
210  p[0] = 1;
211  for (int i = 1; i < (int) p.size(); i++)
212  p[i] = p[i - 1] * 10;
213  vector<int> res;
214  long long cur = 0;
215  int cur_digits = 0;
216  for (int i = 0; i < (int) rhs.size(); i++) {
217  cur += rhs[i] * p[cur_digits];
218  cur_digits += prev;
219  while (cur_digits >= newDigits) {
220  res.push_back(int(cur % p[newDigits]));
221  cur /= p[newDigits];
222  cur_digits -= newDigits;
223  }
224  }
225  res.push_back((int) cur);
226  while (!res.empty() && !res.back())
227  res.pop_back();
228  return res;
